import React, { useState, useEffect } from "react";
import { useRequests } from "../hooks/useRequests";
import { Card } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Link } from "wouter";
import {
  MessageCircle, Check, X, ShieldAlert, Calendar, PlusCircle, Trash2,
  Edit3, AlertTriangle, Bell, Clock, Sparkles, Phone, CheckCircle, Radio
} from "lucide-react";
import {
  Holiday, getHolidays, saveHolidays, getForceStatus, setForceStatus,
  getForceReason, setForceReason, checkShopStatus
} from "../utils/shopStatus";

interface Subscriber {
  number: string;
  timestamp: string;
}

export default function Admin() {
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [password, setPassword] = useState("");
  const { requests, updateRequestStatus } = useRequests();
  const [rejectReason, setRejectReason] = useState<Record<string, string>>({});
  const [holidays, setHolidays] = useState<Holiday[]>(() => getHolidays());
  const [forceStatus, setForceStatusState] = useState<any>(() => getForceStatus());
  const [forceReason, setForceReasonState] = useState(() => getForceReason());
  const [holidayStart, setHolidayStart] = useState("");
  const [holidayEnd, setHolidayEnd] = useState("");
  const [holidayReason, setHolidayReason] = useState("");
  const [editingId, setEditingId] = useState<string | null>(null);
  const [preorders, setPreorders] = useState<any[]>([]);
  const [subscribers, setSubscribers] = useState<Subscriber[]>([]);
  const [orders, setOrders] = useState<any[]>([]);
  const [notifications, setNotifications] = useState<any[]>([]);

  useEffect(() => {
    const loadState = () => {
      try {
        const savedPre = localStorage.getItem("guru_nanak_preorders");
        setPreorders(savedPre ? JSON.parse(savedPre) : []);
      } catch (e) {}
      try {
        const savedSub = localStorage.getItem("guru_nanak_notification_numbers");
        setSubscribers(savedSub ? JSON.parse(savedSub) : []);
      } catch (e) {}
      try {
        const savedOrders = localStorage.getItem("guru_nanak_orders");
        setOrders(savedOrders ? JSON.parse(savedOrders) : []);
      } catch (e) {}
      try {
        const savedNotif = localStorage.getItem("guru_nanak_admin_notifications");
        const list = savedNotif ? JSON.parse(savedNotif) : [];
        if (list.length === 0) {
          const defaultList = [
            { id: "NOTIF_CLOSE_WARN", title: "⏰ SHOP CLOSING SOON", detail: "Time: 7:30PM • Pending Orders: 3", isClosingWarn: true, unread: true },
            { id: "NOTIF_AFTER_HOURS", title: "📱 NEW ENQUIRY (After Hours)", customer: "Rahul", detail: "Wants: 5 Litre Castrol Active 4T • Time: 10:30 PM", isAfterHours: true, unread: true, phone: "9876543210" }
          ];
          localStorage.setItem("guru_nanak_admin_notifications", JSON.stringify(defaultList));
          setNotifications(defaultList);
        } else {
          setNotifications(list);
        }
      } catch (e) {}
    };
    loadState();
    const interval = setInterval(loadState, 2000);
    return () => clearInterval(interval);
  }, []);

  const handleProcessRefund = (orderId: string) => {
    try {
      const savedOrders = localStorage.getItem("guru_nanak_orders");
      if (savedOrders) {
        const list = JSON.parse(savedOrders);
        const updated = list.map((order: any) => order.orderId === orderId ? { ...order, status: "REFUNDED", statusMessage: "Refund processed by owner!" } : order);
        localStorage.setItem("guru_nanak_orders", JSON.stringify(updated));
        setOrders(updated);
      }
      const savedLatest = localStorage.getItem("guru_nanak_latest_order");
      if (savedLatest) {
        const latestObj = JSON.parse(savedLatest);
        if (latestObj.orderId === orderId) {
          latestObj.status = "REFUNDED";
          latestObj.statusMessage = "Refund processed by owner! Your payment will be credited within 5-7 business days.";
          localStorage.setItem("guru_nanak_latest_order", JSON.stringify(latestObj));
        }
      }
      const savedNotifs = localStorage.getItem("guru_nanak_admin_notifications");
      if (savedNotifs) {
        const list = JSON.parse(savedNotifs);
        const filtered = list.filter((n: any) => n.orderId !== orderId);
        localStorage.setItem("guru_nanak_admin_notifications", JSON.stringify(filtered));
        setNotifications(filtered);
      }
      alert(`Refund processed successfully for Order ${orderId}!`);
    } catch (e) { console.error(e); }
  };

  const handleUpdateForceState = (state: "auto" | "open" | "closed") => {
    setForceStatus(state);
    setForceStatusState(state);
    window.dispatchEvent(new Event("storage"));
  };

  const handleUpdateForceReason = (reason: string) => {
    setForceReason(reason);
    setForceReasonState(reason);
  };

  const handleSaveHoliday = (e: React.FormEvent) => {
    e.preventDefault();
    if (!holidayStart || !holidayEnd || !holidayReason) { alert("Please fill all date and reason fields."); return; }
    let updatedList: Holiday[] = [];
    if (editingId) {
      updatedList = holidays.map((h) => h.id === editingId ? { ...h, startDate: holidayStart, endDate: holidayEnd, reason: holidayReason } : h);
      setEditingId(null);
    } else {
      const newHoliday: Holiday = { id: "HOL_" + Date.now(), startDate: holidayStart, endDate: holidayEnd, reason: holidayReason };
      updatedList = [...holidays, newHoliday];
    }
    setHolidays(updatedList);
    saveHolidays(updatedList);
    setHolidayStart(""); setHolidayEnd(""); setHolidayReason("");
    window.dispatchEvent(new Event("storage"));
  };

  const handleDeleteHoliday = (id: string) => {
    const updated = holidays.filter((h) => h.id !== id);
    setHolidays(updated); saveHolidays(updated);
    window.dispatchEvent(new Event("storage"));
  };

  const handleEditHoliday = (h: Holiday) => {
    setEditingId(h.id); setHolidayStart(h.startDate); setHolidayEnd(h.endDate); setHolidayReason(h.reason);
  };

  const handlePreOrderAction = (id: string, action: "CONFIRMED" | "REJECTED") => {
    try {
      const updated = preorders.map((po) => po.id === id ? { ...po, status: action } : po);
      setPreorders(updated);
      localStorage.setItem("guru_nanak_preorders", JSON.stringify(updated));
    } catch (e) { console.error(e); }
  };

  const handleDismissNotification = (id: string) => {
    try {
      const updated = notifications.filter((n) => n.id !== id);
      setNotifications(updated);
      localStorage.setItem("guru_nanak_admin_notifications", JSON.stringify(updated));
    } catch (e) {}
  };

  const handleWhatsAppTrigger = (phone: string, text: string) => {
    window.open(`https://wa.me/91${phone}?text=${encodeURIComponent(text)}`, "_blank");
  };

  const handleWhatsAppBroadcast = () => {
    const message = encodeURIComponent("Guru Nanak Oil Agency: Shop ab OPEN hai! Aapke orders process ho rahe hain.");
    subscribers.forEach((sub) => { window.open(`https://wa.me/91${sub.number}?text=${message}`, "_blank"); });
  };

  const currentStatusObject = checkShopStatus();
  const todayFormat = new Date().toLocaleDateString("en-US", { day: "numeric", month: "long", year: "numeric" });

  if (!isAuthenticated) {
    return (
      <div className="min-h-screen bg-slate-50 flex items-center justify-center p-4">
        <Card className="max-w-md w-full p-8 shadow-xl">
          <h2 className="text-2xl font-black text-[#0a2351] mb-6 text-center">Admin Login</h2>
          <form onSubmit={(e) => { e.preventDefault(); if (password === "admin123") setIsAuthenticated(true); else alert("Incorrect password"); }} className="space-y-4">
            <input type="password" value={password} onChange={(e) => setPassword(e.target.value)} placeholder="Enter Password" className="w-full bg-slate-50 px-4 py-3 rounded-xl border border-gray-200 focus:border-[#0a2351] focus:ring-2 focus:ring-[#0a2351]/20 outline-none transition-all" />
            <Button type="submit" className="w-full bg-amber-500 hover:bg-amber-600 text-[#0a2351] font-black">Login</Button>
            <Link href="/"><Button type="button" variant="outline" className="w-full mt-2">Back to Store</Button></Link>
          </form>
        </Card>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-slate-50 pb-20">
      <div className="bg-slate-900 text-white pt-10 pb-16 px-4">
        <div className="container max-w-7xl mx-auto flex flex-col md:flex-row justify-between items-start md:items-center gap-6">
          <div>
            <div className="flex items-center gap-3">
              <ShieldAlert className="w-9 h-9 text-amber-500 shrink-0" />
              <div>
                <h1 className="text-3xl md:text-5xl font-black uppercase tracking-tight">Admin Panel</h1>
                <p className="text-slate-400 font-medium mt-1">Configure business hours, holidays & orders</p>
              </div>
            </div>
          </div>
          <Link href="/"><Button className="bg-[#0a2351] hover:bg-amber-500 hover:text-[#0a2351] text-white font-bold">← Return to Online Store</Button></Link>
        </div>
      </div>

      <div className="container max-w-7xl mx-auto px-4 -mt-8 grid lg:grid-cols-12 gap-6">
        <div className="lg:col-span-8 space-y-6">
          <div className="bg-white rounded-3xl shadow-sm border border-slate-200 p-6">
            <div className="flex items-center justify-between border-b border-slate-100 pb-4 mb-4">
              <div className="flex items-center gap-2"><Clock className="w-5 h-5 text-[#0a2351]" /><h2 className="text-lg font-black text-slate-800 uppercase tracking-tight">Shop Status Control</h2></div>
              <span className="text-xs font-bold text-slate-500 bg-slate-100 px-3 py-1 rounded-full">{todayFormat}</span>
            </div>
            <div className="grid md:grid-cols-2 gap-6 items-center">
              <div>
                <h3 className="text-xs font-black uppercase text-slate-400 tracking-wider mb-2">Current Live Status</h3>
                <div className="flex items-center gap-3 mt-2">
                  <div className={`w-4 h-4 rounded-full animate-ping ${currentStatusObject.isOpen ? "bg-emerald-500" : "bg-rose-500"}`} />
                  <span className={`text-2xl font-black uppercase ${currentStatusObject.isOpen ? "text-emerald-600" : "text-rose-600"}`}>{currentStatusObject.isOpen ? "🟢 OPEN" : "🔴 CLOSED"}</span>
                </div>
                <p className="text-xs text-slate-500 font-bold mt-2">Status evaluated automatically based on standard business hours</p>
              </div>
              <div className="p-4 bg-slate-50 rounded-2xl border border-slate-200">
                <h4 className="text-xs font-black uppercase tracking-wider text-slate-500 mb-3">Force Override</h4>
                <div className="grid grid-cols-3 gap-2">
                  <button onClick={() => handleUpdateForceState("open")} className={`py-2 px-1 rounded-xl text-xs font-extrabold transition-colors ${forceStatus === "open" ? "bg-emerald-500 text-white" : "bg-white border border-slate-200 text-slate-600 hover:bg-slate-100"}`}>Force Open</button>
                  <button onClick={() => handleUpdateForceState("closed")} className={`py-2 px-1 rounded-xl text-xs font-extrabold transition-colors ${forceStatus === "closed" ? "bg-rose-500 text-white" : "bg-white border border-slate-200 text-slate-600 hover:bg-slate-100"}`}>Force Close</button>
                  <button onClick={() => handleUpdateForceState("auto")} className={`py-2 px-1 rounded-xl text-xs font-extrabold transition-colors ${forceStatus === "auto" ? "bg-[#0a2351] text-white" : "bg-white border border-slate-200 text-slate-600 hover:bg-slate-100"}`}>Auto Timer</button>
                </div>
              </div>
            </div>
            <div className="mt-6 pt-4 border-t border-slate-100">
              <label className="block text-xs font-black text-slate-500 uppercase tracking-wider mb-2">Custom Force Closed Reason (visible to customers)</label>
              <div className="flex gap-2">
                <input type="text" value={forceReason} onChange={(e) => handleUpdateForceReason(e.target.value)} placeholder="e.g. Diwali Holidays - Back on 5th Oct" className="flex-1 bg-slate-50 px-4 py-2.5 rounded-xl text-sm font-bold border border-slate-200 focus:border-[#0a2351] outline-none" />
              </div>
              <p className="text-[10px] text-slate-400 font-semibold mt-1">This message will be shown to customers when shop is force-closed</p>
            </div>
          </div>

          <div className="bg-white rounded-3xl shadow-sm border border-slate-200 p-6">
            <h2 className="text-lg font-black uppercase text-slate-800 tracking-tight flex items-center gap-2 mb-4"><Calendar className="w-5 h-5 text-amber-500" /> Holiday Calendar</h2>
            <div className="grid md:grid-cols-2 gap-8">
              <form onSubmit={handleSaveHoliday} className="space-y-4">
                <h3 className="text-xs font-black uppercase tracking-wider text-slate-500 flex items-center gap-1"><PlusCircle className="w-4 h-4 text-[#0a2351]" />{editingId ? "Edit Holiday" : "Add Holiday"}</h3>
                <div className="grid grid-cols-2 gap-3">
                  <div><label className="block text-[10px] font-black uppercase text-slate-400 mb-1">Start Date</label><input type="date" required value={holidayStart} onChange={(e) => setHolidayStart(e.target.value)} className="w-full bg-slate-50 px-3 py-2 rounded-xl text-sm font-bold border border-slate-200 focus:border-[#0a2351] outline-none" /></div>
                  <div><label className="block text-[10px] font-black uppercase text-slate-400 mb-1">End Date</label><input type="date" required value={holidayEnd} onChange={(e) => setHolidayEnd(e.target.value)} className="w-full bg-slate-50 px-3 py-2 rounded-xl text-sm font-bold border border-slate-200 focus:border-[#0a2351] outline-none" /></div>
                </div>
                <div><label className="block text-[10px] font-black uppercase text-slate-400 mb-1">Reason</label><input type="text" required value={holidayReason} onChange={(e) => setHolidayReason(e.target.value)} placeholder="e.g. Eid Festival, Diwali Break" className="w-full bg-slate-50 px-3 py-2 rounded-xl text-sm font-bold border border-slate-200 focus:border-[#0a2351] outline-none" /></div>
                <div className="flex gap-2">
                  <button type="submit" className="flex-1 py-2.5 bg-[#0a2351] hover:bg-amber-500 hover:text-[#0a2351] text-white font-black rounded-xl transition-colors text-sm">{editingId ? "Save Edits" : "Schedule Holiday"}</button>
                  {editingId && <button type="button" onClick={() => { setEditingId(null); setHolidayStart(""); setHolidayEnd(""); setHolidayReason(""); }} className="px-3 bg-slate-100 hover:bg-slate-200 rounded-xl text-sm font-bold">Cancel</button>}
                </div>
              </form>
              <div>
                <h3 className="text-xs font-black uppercase tracking-wider text-slate-500 mb-3">Scheduled Holidays</h3>
                {holidays.length === 0 ? <p className="text-xs font-bold text-slate-400 py-6 text-center">No holidays scheduled</p> : (
                  <div className="space-y-2 max-h-56 overflow-y-auto pr-1">
                    {holidays.map((h) => (
                      <div key={h.id} className="p-3 bg-[#0a2351]/5 border border-slate-200 rounded-xl flex items-center gap-3">
                        <div className="flex-1">
                          <p className="text-xs font-extrabold text-[#0a2351]">{h.reason}</p>
                          <p className="text-[10px] text-slate-500 font-bold">📅 {new Date(h.startDate).toLocaleDateString("en-US")}{h.startDate !== h.endDate && ` to ${new Date(h.endDate).toLocaleDateString("en-US")}`}</p>
                        </div>
                        <div className="flex gap-1 shrink-0">
                          <button onClick={() => handleEditHoliday(h)} className="p-1.5 bg-slate-100 hover:bg-slate-200 rounded-lg transition-colors" title="Edit Holiday"><Edit3 className="w-3.5 h-3.5" /></button>
                          <button onClick={() => handleDeleteHoliday(h.id)} className="p-1.5 bg-rose-50 hover:bg-rose-100 text-rose-600 rounded-lg transition-colors" title="Delete Holiday"><Trash2 className="w-3.5 h-3.5" /></button>
                        </div>
                      </div>
                    ))}
                  </div>
                )}
              </div>
            </div>
          </div>

          <div className="bg-white rounded-3xl shadow-sm border border-slate-200 p-6">
            <h2 className="text-lg font-black uppercase text-slate-800 tracking-tight flex items-center gap-2 mb-4"><Sparkles className="w-5 h-5 text-amber-500" /> Pending Pre-Orders</h2>
            {preorders.length === 0 ? <div className="py-12 text-center text-slate-400 font-bold text-sm">No customer pre-order reservations available.</div> : (
              <div className="space-y-3">
                {preorders.map((po) => (
                  <div key={po.id} className="border border-slate-200 rounded-xl p-4 flex flex-col md:flex-row justify-between items-start md:items-center gap-4">
                    <div>
                      <div className="flex items-center gap-2 flex-wrap"><span className="text-[10px] bg-amber-100 text-amber-700 px-2 py-0.5 rounded font-black">{po.id}</span><span className="text-xs text-slate-400 font-bold">{new Date(po.date).toLocaleDateString()}</span></div>
                      <h4 className="font-extrabold text-sm text-[#0a2351] mt-1">{po.quantity} {po.product}</h4>
                      <p className="text-xs text-slate-600 font-bold mt-1">Customer: <strong className="text-slate-800">{po.name}</strong> • {po.phone}</p>
                      {po.notes && <p className="text-xs bg-slate-50 italic text-slate-500 mt-1 p-2 rounded">Note: "{po.notes}"</p>}
                    </div>
                    <div className="flex items-center gap-2 shrink-0 w-full md:w-auto">
                      {po.status === "PENDING_CONFIRMATION" ? (
                        <>
                          <button onClick={() => handlePreOrderAction(po.id, "CONFIRMED")} className="flex-1 md:flex-none py-1.5 px-3.5 bg-emerald-500 hover:bg-emerald-600 text-white rounded-lg font-bold text-xs flex items-center gap-1"><Check className="w-3.5 h-3.5" /> Approve</button>
                          <button onClick={() => handlePreOrderAction(po.id, "REJECTED")} className="flex-1 md:flex-none py-1.5 px-3.5 bg-rose-500 hover:bg-rose-600 text-white rounded-lg font-bold text-xs flex items-center gap-1"><X className="w-3.5 h-3.5" /> Decline</button>
                        </>
                      ) : (
                        <div className="flex items-center gap-2 w-full md:w-auto">
                          <span className={`text-[10px] font-black px-2.5 py-1 rounded-full ${po.status === "CONFIRMED" ? "bg-emerald-100 text-emerald-700" : "bg-rose-100 text-rose-700"}`}>{po.status}</span>
                          <button onClick={() => handleWhatsAppTrigger(po.phone, `Your pre-order ${po.id} has been ${po.status.toLowerCase()}.`)} className="py-1.5 px-3.5 bg-emerald-100 hover:bg-emerald-200 text-emerald-700 rounded-lg font-bold text-xs flex items-center gap-1"><MessageCircle className="w-3.5 h-3.5 shrink-0" /> Notify</button>
                        </div>
                      )}
                    </div>
                  </div>
                ))}
              </div>
            )}
          </div>

          <div className="bg-white rounded-3xl shadow-sm border border-slate-200 p-6">
            <h2 className="text-lg font-black uppercase text-slate-800 tracking-tight flex items-center gap-2 mb-4"><PlusCircle className="w-5 h-5 text-amber-500" /> 📦 Store Bookings & Refunds</h2>
            {orders.length === 0 ? <div className="py-8 text-center text-slate-400 font-bold text-sm">No store orders or payments logged yet.</div> : (
              <div className="space-y-4">
                {orders.map((order: any, idx: number) => {
                  const isCancelRequested = order.status === "CANCELLED_REQUESTED";
                  const isRefunded = order.status === "REFUNDED";
                  return (
                    <div key={order.orderId || idx} className={`border rounded-2xl p-4 transition-all ${isCancelRequested ? "border-red-400 bg-red-50/20" : isRefunded ? "border-gray-200 bg-gray-50/50" : "border-slate-200 bg-white"}`}>
                      <div className="flex justify-between items-start flex-wrap gap-2 mb-3">
                        <div>
                          <span className={`text-[10px] font-black px-2.5 py-1 rounded-full ${isCancelRequested ? "bg-rose-100 text-rose-800" : isRefunded ? "bg-gray-100 text-gray-700" : order.isScheduled ? "bg-amber-100 text-amber-800" : "bg-emerald-100 text-emerald-800"}`}>{isCancelRequested ? "⚠️ REFUND REQUEST" : isRefunded ? "Ref: REFUNDED" : order.isScheduled ? "⏰ Auto-Scheduled Booked" : "📦 Standard"}</span>
                          <span className="text-xs text-slate-400 font-bold ml-2">{order.orderId}</span>
                        </div>
                        <span className="text-xs font-black text-[#0a2351]">₹{order.grandTotal?.toLocaleString("en-IN")} ({order.items?.length || 0} items)</span>
                      </div>
                      <div className="space-y-1.5 text-xs text-slate-600 font-medium">
                        <p><strong>Customer Address:</strong> {order.address}</p>
                        <p><strong>Items:</strong> {order.items?.map((it: any) => `${it.quantity}x ${it.name}`).join(", ")}</p>
                        <p className="text-slate-400 text-[10px]">Ordered on: {new Date(order.date).toLocaleString()}</p>
                      </div>
                      {isCancelRequested && (
                        <div className="mt-4 bg-red-50 border border-red-200 rounded-xl p-3">
                          <p className="text-xs font-extrabold text-red-800">🚨 Customer Cancelled Order</p>
                          <p className="text-[11px] text-gray-700 font-bold mt-1">Customer ne order cancel kiya hai. Mode: {order.paymentMethod}</p>
                          <div className="pt-1"><button onClick={() => handleProcessRefund(order.orderId)} className="w-full py-2 bg-red-600 hover:bg-red-700 text-white rounded-lg font-bold">💸 Process Refund & Close Refund Task</button></div>
                        </div>
                      )}
                      {isRefunded && <div className="mt-2.5 bg-gray-100 p-2.5 rounded-xl text-xs font-bold text-gray-600">✅ REFUND FULLY PROCESSED & ORDER CLOSED</div>}
                      {!isCancelRequested && !isRefunded && (
                        <div className="mt-3 flex justify-between items-center">
                          <span className="text-[10px] uppercase font-bold text-slate-500">Status: <strong className="text-emerald-700">{order.status || "ACTIVE"}</strong></span>
                          <button onClick={() => alert(`Order ${order.orderId} marked as dispatched!`)} className="text-[9px] font-black uppercase text-[#0a2351] bg-amber-100 hover:bg-amber-200 px-3 py-1.5 rounded-lg transition-colors">Mark Dispatched</button>
                        </div>
                      )}
                    </div>
                  );
                })}
              </div>
            )}
          </div>
        </div>

        <div className="lg:col-span-4 space-y-6">
          <div className="bg-white rounded-3xl shadow-sm border border-slate-200 p-6">
            <div className="flex items-center justify-between border-b border-slate-100 pb-4 mb-4">
              <div className="flex items-center gap-2"><Bell className="w-5 h-5 text-amber-500 animate-swing" /><h3 className="font-black text-slate-800 uppercase tracking-tight text-sm">Live Alerts</h3></div>
              <span className="bg-rose-500 text-white font-bold font-mono text-xs px-2 py-1 rounded-full">{notifications.length}</span>
            </div>
            <div className="space-y-3">
              {notifications.length === 0 ? <div className="text-center text-xs text-slate-400 font-semibold py-4">No alerts in owner inbox.</div> : (
                notifications.map((n) => (
                  <div key={n.id} className="p-4 bg-slate-50 border border-slate-200 rounded-xl relative">
                    <button onClick={() => handleDismissNotification(n.id)} className="absolute top-3 right-3 text-slate-300 hover:text-rose-500 transition-colors">×</button>
                    <div className="space-y-2">
                      <div className="flex items-center gap-1.5"><Radio className="w-3.5 h-3.5 text-rose-500 shrink-0" /><span className="font-extrabold text-xs text-slate-800">{n.title}</span></div>
                      <p className="text-xs font-semibold text-slate-600 leading-relaxed">{n.detail ? n.detail.replace(/&bull;/g, "•") : ""}</p>
                      {n.isClosingWarn && (
                        <div className="bg-amber-100 border border-amber-200 rounded-lg p-2 mt-2">
                          <p className="text-[10px] font-bold text-amber-800">⏰ Time is past 7:30PM — Pack dispatch?</p>
                          <div className="flex gap-1.5 mt-1">
                            <button onClick={() => { handleDismissNotification(n.id); alert("Dispatch packing sequence triggered!"); }} className="px-2 py-1 bg-amber-500 hover:bg-amber-600 text-white text-[10px] font-bold rounded">Yes, Pack</button>
                            <button onClick={() => { handleDismissNotification(n.id); alert("Scheduled dispatch shifted automatically to tomorrow!"); }} className="px-2 py-1 bg-slate-200 hover:bg-slate-300 text-slate-700 text-[10px] font-bold rounded">Dispatch Tomorrow</button>
                          </div>
                        </div>
                      )}
                      {n.isAfterHours && (
                        <div className="grid grid-cols-3 gap-1 pt-1">
                          <button onClick={() => handleWhatsAppTrigger(n.phone, "Thank you for contacting Guru Nanak Oil Agency. We will assist you tomorrow at 9 AM.")} className="bg-emerald-600 hover:bg-emerald-700 text-white text-[10px] font-bold py-1.5 rounded">💬 WhatsApp</button>
                          <a href={`tel:+91${n.phone}`} className="bg-sky-600 hover:bg-sky-700 text-white text-[10px] font-bold py-1.5 rounded text-center">📞 Call</a>
                          <button onClick={() => { handleDismissNotification(n.id); alert("Remind Task configured for tomorrow at 9 AM!"); }} className="bg-amber-500 text-[#0a2351] text-[10px] font-bold py-1.5 rounded">⏰ Remind Tomorrow</button>
                        </div>
                      )}
                      {!n.isClosingWarn && !n.isAfterHours && <div className="bg-[#0a2351]/5 hover:bg-[#0a2351]/10 rounded-lg p-2 text-[10px] text-slate-500 font-medium cursor-pointer transition-colors">Ref: {n.id} • Customer: {n.customer || "Guest"}</div>}
                    </div>
                  </div>
                ))
              )}
            </div>
          </div>

          <div className="bg-white rounded-3xl shadow-sm border border-slate-200 p-6">
            <h3 className="font-black text-slate-800 uppercase tracking-tight text-sm flex items-center gap-2 mb-3"><MessageCircle className="w-5 h-5 text-emerald-500" /> WhatsApp Subscribers</h3>
            <p className="text-xs text-gray-500 font-bold leading-normal mb-3">Client mobile numbers registered for "Shop Open" alerts.</p>
            {subscribers.length === 0 ? <p className="text-xs font-bold text-slate-400 py-6 text-center">No subscribers yet</p> : (
              <div className="space-y-2">
                {subscribers.map((sub, i) => (
                  <div key={i} className="p-2.5 bg-slate-50 rounded-xl flex items-center justify-between">
                    <div><span className="font-mono font-black text-slate-800 text-sm">+91 {sub.number}</span><span className="text-[9px] block text-slate-400 font-medium">{new Date(sub.timestamp).toLocaleDateString()}</span></div>
                    <button onClick={() => handleWhatsAppTrigger(sub.number, "Guru Nanak Oil Agency is now OPEN! Visit us or place your order online.")} className="py-1 px-2.5 bg-emerald-600 hover:bg-emerald-700 text-white text-[10px] font-bold rounded-lg transition-colors">Notify</button>
                  </div>
                ))}
                <button onClick={handleWhatsAppBroadcast} className="w-full py-2 bg-emerald-600 hover:bg-emerald-700 text-white rounded-lg font-bold flex items-center justify-center gap-2 mt-3">🚀 Send Broadcast</button>
              </div>
            )}
          </div>
        </div>
      </div>
    </div>
  );
}
