import React, { useState, useEffect } from "react";
import { ShoppingCart, Bell, Send, AlertTriangle, Clock } from "lucide-react";
import { Button } from "./ui/button";
import { useCart } from "../hooks/useCart";
import { useRequests } from "../hooks/useRequests";
import { useLocation } from "wouter";
import { checkShopStatus } from "../utils/shopStatus";

interface ProductAddToCartProps {
  product: {
    id: string;
    name: string;
    price?: number;
    stock_quantity: number;
    image?: string;
    unit?: string;
    vehicleCategory?: string;
    grades?: {
      name: string;
      price: number;
      packaging?: { size: string; price?: number }[];
    }[];
  };
}

export function ProductAddToCart({ product }: ProductAddToCartProps) {
  const { cart, addToCart } = useCart();
  const { addRequest } = useRequests();
  const [, setLocation] = useLocation();
  const [selectedGrade, setSelectedGrade] = useState<string>("");
  const [selectedPackaging, setSelectedPackaging] = useState<string>("");
  const [shopStatus, setShopStatus] = useState(() => checkShopStatus());

  useEffect(() => {
    const timer = setInterval(() => {
      setShopStatus(checkShopStatus());
    }, 2000);
    return () => clearInterval(timer);
  }, []);

  const isFromPreorder = typeof window !== "undefined" && window.location.search.includes("fromPreorder=true");
  const [preorderQty, setPreorderQty] = useState(0);
  const { isOpen } = shopStatus;

  const stock = product.stock_quantity || 0;
  const unit = product.unit || "Litre";

  const currentGradeObj = product.grades?.find((g) => g.name === selectedGrade);

  const isTwoWheeler =
    product.vehicleCategory === "Two Wheeler" ||
    product.name?.toLowerCase().includes("activ") ||
    product.name?.toLowerCase().includes("power1") ||
    product.name?.toLowerCase().includes("scooter");

  const packagingOptions =
    currentGradeObj?.packaging && currentGradeObj.packaging.length > 0
      ? currentGradeObj.packaging
      : isTwoWheeler
      ? [{ size: "1LTR" }]
      : [{ size: "Standard (1 Unit)" }];

  useEffect(() => {
    if (currentGradeObj) {
      const actualPackagingList =
        currentGradeObj.packaging && currentGradeObj.packaging.length > 0
          ? currentGradeObj.packaging
          : isTwoWheeler
          ? [{ size: "1LTR" }]
          : [{ size: "Standard (1 Unit)" }];

      const optionExists = actualPackagingList.find((p) => p.size === selectedPackaging);
      if (!optionExists && actualPackagingList.length > 0) {
        setSelectedPackaging(actualPackagingList[0].size);
      } else if (actualPackagingList.length === 0) {
        setSelectedPackaging("Standard (1 Unit)");
      }
    }
  }, [selectedGrade, currentGradeObj, isTwoWheeler, selectedPackaging]);

  let itemPrice = product.price || 0;
  if (product.grades && product.grades.length > 0) {
    if (selectedGrade) {
      const matchGrade = product.grades.find((g) => g.name === selectedGrade);
      if (matchGrade) {
        const actualPackagingList =
          matchGrade.packaging && matchGrade.packaging.length > 0
            ? matchGrade.packaging
            : isTwoWheeler
            ? [{ size: "1LTR" }]
            : [{ size: "Standard (1 Unit)" }];

        if (actualPackagingList && selectedPackaging) {
          const packObj = actualPackagingList.find((p) => p.size === selectedPackaging);
          itemPrice = packObj?.price || matchGrade.price;
        } else {
          itemPrice = matchGrade.price;
        }
      }
    } else {
      itemPrice = product.grades[0].price;
    }
  }

  useEffect(() => {
    if (isFromPreorder && typeof window !== "undefined") {
      const updateQty = () => {
        try {
          const existingJson = localStorage.getItem("preorder_selected_items");
          const items = existingJson ? JSON.parse(existingJson) : [];
          const variantNamePart = Array.from(new Set([selectedGrade, selectedPackaging]))
            .filter(Boolean)
            .join(", ");
          const variantId = selectedGrade
            ? `${product.id}-${selectedGrade.replace(/\s/g, "")}`
            : product.id;
          const match = items.find((it: any) => it.id === variantId);
          setPreorderQty(match ? match.quantity : 0);
        } catch (e) {
          console.error("Error parsing preorder items", e);
        }
      };
      updateQty();
      window.addEventListener("preorder_items_updated", updateQty);
      return () => window.removeEventListener("preorder_items_updated", updateQty);
    }
  }, [selectedGrade, selectedPackaging, product.id, isFromPreorder]);

  const handleUpdatePreorderQty = (change: number) => {
    if (typeof window === "undefined") return;

    const variantNamePart = Array.from(new Set([selectedGrade, selectedPackaging]))
      .filter(Boolean)
      .join(", ");
    const variantId = selectedGrade ? `${product.id}-${selectedGrade.replace(/\s/g, "")}` : product.id;
    const finalName = `${product.name}${variantNamePart ? ` (${variantNamePart})` : ""}`;

    try {
      const existingJson = localStorage.getItem("preorder_selected_items");
      let items = existingJson ? JSON.parse(existingJson) : [];
      const idx = items.findIndex((it: any) => it.id === variantId);

      if (idx > -1) {
        items[idx].quantity += change;
        if (items[idx].quantity <= 0) {
          items.splice(idx, 1);
        }
      } else if (change > 0) {
        items.push({
          id: variantId,
          name: finalName,
          price: itemPrice,
          quantity: change,
          unit: unit,
          image: product.image,
        });
      }

      localStorage.setItem("preorder_selected_items", JSON.stringify(items));
      window.dispatchEvent(new Event("preorder_items_updated"));
    } catch (e) {
      console.error("Error updating preorder items", e);
    }
  };

  const quantity = 1;
  const totalPrice = itemPrice * quantity;
  const isOutOfStockQuantity = quantity > stock;
  const needsOwnerApproval = isOutOfStockQuantity;

  const validGrade = product.grades ? selectedGrade !== "" && selectedPackaging !== "" : true;

  const handleAdd = () => {
    if (!validGrade) return;

    if (needsOwnerApproval) {
      addRequest({
        productId: product.id,
        productName: product.name,
        requestedGrade: selectedGrade || "N/A",
        requestedQuantity: quantity,
        totalPrice: totalPrice,
        status: "WAITING FOR OWNER",
        unit: unit,
      });
      setLocation("/status");
    } else {
      const variantNamePart = Array.from(new Set([selectedGrade, selectedPackaging]))
        .filter(Boolean)
        .join(", ");
      const variantId = selectedGrade ? `${product.id}-${selectedGrade.replace(/\s/g, "")}` : product.id;

      addToCart(
        {
          id: variantId,
          name: `${product.name}${variantNamePart ? ` (${variantNamePart})` : ""}`,
          price: itemPrice,
          stock_quantity: stock,
          image: product.image,
          unit: unit,
        },
        quantity
      );

      if (product.grades) {
        setSelectedGrade("");
        setSelectedPackaging("");
      }
    }
  };

  return (
    <div className="bg-white border-2 border-gray-100 rounded-xl p-5 shadow-sm">
      {product.grades && product.grades.length > 0 && (
        <div className="space-y-3 mb-4">
          <label className="text-xs font-black text-gray-800 uppercase tracking-wider block">
            🛢️ SELECT GRADE (Oil Number):
          </label>
          <div className="grid grid-cols-3 gap-2">
            {product.grades.map((g) => (
              <button
                key={g.name}
                onClick={() => setSelectedGrade(g.name)}
                className={`py-3 px-1 rounded-lg border-2 flex flex-col items-center justify-center transition-all ${
                  selectedGrade === g.name
                    ? "border-[#0a2351] bg-blue-50 text-[#0a2351] shadow-inner"
                    : "border-gray-200 text-gray-600 hover:border-gray-300"
                }`}
              >
                <span className="font-bold text-sm tracking-tight">{g.name}</span>
                <span className="text-xs font-black mt-1 text-gray-400">₹{g.price}</span>
              </button>
            ))}
          </div>
        </div>
      )}

      {selectedGrade && packagingOptions && packagingOptions.length > 0 && (
        <div className="space-y-3 mb-4">
          <label className="text-xs font-black text-gray-800 uppercase tracking-wider block">
            📦 SELECT PACKAGING:
          </label>
          <div className="grid grid-cols-3 gap-2">
            {packagingOptions.map((pack) => (
              <button
                key={pack.size}
                onClick={() => setSelectedPackaging(pack.size)}
                className={`py-3 px-1 rounded-lg border-2 flex flex-col items-center justify-center transition-all ${
                  selectedPackaging === pack.size
                    ? "border-[#0a2351] bg-blue-50 text-[#0a2351] shadow-inner"
                    : "border-gray-200 text-gray-600 hover:border-gray-300"
                }`}
              >
                <span className="font-bold text-sm tracking-tight">{pack.size}</span>
                {pack.price && <span className="text-xs font-black mt-1 text-gray-400">₹{pack.price}</span>}
              </button>
            ))}
          </div>
        </div>
      )}

      <div className="bg-gray-50 p-4 rounded-xl border border-gray-200 space-y-2 mb-4">
        <div className="flex justify-between items-end">
          <div className="flex flex-col">
            <span className="text-sm text-gray-600 font-bold tracking-wide">
              💰 TOTAL: ₹{totalPrice.toLocaleString("en-IN")}
            </span>
            <span className="text-[11px] text-gray-400 font-bold tracking-wide">per {unit}</span>
          </div>
          <div className="flex flex-col items-end text-right">
            <span className="text-sm text-amber-600 font-bold tracking-wide">🚚 Delivery: Calculated at Checkout</span>
          </div>
        </div>
        <div className="pt-3 border-t border-gray-200 flex justify-between items-center">
          <span className="text-sm font-black tracking-widest uppercase text-gray-500">Final Price:</span>
          <span className="text-2xl font-black text-[#0a2351]">₹{totalPrice.toLocaleString("en-IN")}</span>
        </div>
      </div>

      <div className="pt-2">
        {needsOwnerApproval ? (
          <div className="bg-orange-50 border-2 border-orange-200 rounded-xl p-4 space-y-3">
            <div className="flex gap-3">
              <AlertTriangle className="w-6 h-6 text-orange-600 shrink-0" />
              <div className="flex flex-col space-y-1">
                <span className="font-black text-orange-800 uppercase tracking-wide text-xs">Low Stock Alert</span>
                <p className="text-xs text-orange-700 font-medium leading-relaxed">
                  You selected <strong>{selectedGrade || "Unknown Grade"}</strong>. Current stock: {stock} units.
                  Available Grades: {product.grades?.map((g) => g.name).join(", ")}
                </p>
              </div>
            </div>
            <Button
              onClick={handleAdd}
              className="w-full h-14 bg-orange-600 hover:bg-orange-700 text-white font-bold"
            >
              <Send className="w-5 h-5 mr-2" />
              Send Request to Owner
            </Button>
            <p className="text-[10px] text-orange-600/80 text-center font-medium">
              Shop owner will check and confirm if this can be arranged.
            </p>
            <Button
              variant="outline"
              className="w-full text-orange-800 border-orange-300"
              onClick={() => {
                if (product.grades) setSelectedGrade(product.grades[0].name);
              }}
            >
              Try Different Options
            </Button>
          </div>
        ) : isFromPreorder ? (
          <div>
            <Button
              onClick={() => handleUpdatePreorderQty(1)}
              disabled={!validGrade}
              className={`w-full font-black h-14 shadow-xl uppercase tracking-wider ${
                !validGrade
                  ? "bg-gray-200 text-gray-500 cursor-not-allowed"
                  : preorderQty > 0
                  ? "bg-emerald-600 hover:bg-emerald-700 text-white"
                  : "bg-amber-500 hover:bg-amber-600 text-[#0a2351]"
              }`}
            >
              {preorderQty > 0 ? (
                <>
                  <span>✓ Added to Pre-Order</span>
                  <span className="text-xs bg-white/25 px-2 py-0.5 rounded ml-2">{preorderQty}</span>
                </>
              ) : (
                "📋 Add to Pre-Order"
              )}
            </Button>
          </div>
        ) : (
          <Button
            onClick={handleAdd}
            disabled={!validGrade}
            className={`w-full font-black h-14 shadow-xl uppercase tracking-wider ${
              !validGrade
                ? "bg-gray-200 text-gray-500 cursor-not-allowed"
                : "bg-[#0a2351] hover:bg-amber-500 hover:text-[#0a2351] text-white"
            }`}
          >
            <ShoppingCart className="w-5 h-5 mr-2" />
            Add to Cart
          </Button>
        )}
      </div>

      {!isOpen && (
        <div className="mt-3 bg-amber-50 border border-amber-200/60 rounded-lg p-3 flex gap-3">
          <Clock className="w-4 h-4 shrink-0 mt-0.5 text-amber-600" />
          <div>
            <span className="font-extrabold block text-amber-900">🕒 SHOP CURRENTLY CLOSED</span>
            <span className="font-semibold text-gray-600 block mt-0.5 text-xs">
              You can still book your order! Select <strong>Auto-Scheduled Delivery</strong> at checkout.
            </span>
          </div>
        </div>
      )}
    </div>
  );
}
