export function formatCountdown(seconds: number): string {
  const safeSeconds = Math.max(0, seconds);
  const hrs = Math.floor(safeSeconds / 3600);
  const mins = Math.floor((safeSeconds % 3600) / 60);
  const secs = safeSeconds % 60;
  return [hrs, mins, secs].map((v) => String(v).padStart(2, "0")).join(":");
}

export function ShopStatusBar() {

  useEffect(() => {

      setStatus(checkShopStatus());
    }, 1000);
    return () => clearInterval(interval);
  }, []);

  const { isOpen, countdownSeconds, nextOpenStr, type, holidayReason } = status;

  return (
    <div
      className={`w-full text-white text-xs md:text-sm font-bold border-b ${
        isOpen
          ? "bg-emerald-600 border-emerald-500"
          : type === "holiday"
          ? "bg-amber-600 border-amber-500"
          : "bg-[#0a2351] border-amber-500/30"
      }`}
    >
      <div className="container max-w-7xl mx-auto px-4 py-3 flex flex-col md:flex-row items-center justify-between gap-2">
        <div className="flex items-center gap-2 flex-wrap justify-center md:justify-start">
          <Clock className={`w-4 h-4 shrink-0 animate-pulse ${isOpen ? "text-emerald-300" : "text-amber-400"}`} />
          <span>
            {isOpen ? (
              <span className="inline-flex items-center gap-1.5">
                <span className="w-2.5 h-2.5 rounded-full bg-emerald-300 animate-pulse" />
                <strong>GURU NANAK OIL: OPEN NOW</strong> (9:00 AM - 8:00 PM)
              </span>

              <span>
                <strong>🔴 FESTIVAL BREAK:</strong> {holidayReason || "Closed for Holiday"}
              </span>
            ) : type === "sunday" ? (
              <span>
                <strong>🔴 SUNDAY CLOSED:</strong> We reopen tomorrow 9:00 AM
              </span>
            ) : (
              <span>
                <strong>🔴 SHOP CLOSED:</strong> Next Open: <strong>{nextOpenStr}</strong>
              </span>
            )}
          </span>
        </div>
        <div className="flex items-center gap-4 flex-wrap justify-center">
          {!isOpen && countdownSeconds > 0 && (
            <span className="font-mono bg-black/20 px-2.5 py-1 rounded text-white">

        </div>
      </div>

      {openPreorder && <PreOrderModal onClose={() => setOpenPreorder(false)} />}
    </div>
  );
}
